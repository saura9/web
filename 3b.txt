<body>
  <input type="text" id="inputString">
  <button id="sortButton">Sort</button>
 

  <script>
    
function Sortstr(){
  var input= document.getElementById("inputString").value;
  var SortedStr=input.split('').sort().join('');
  alert("Sorted string: "+SortedStr);
}
document.getElementById("sortButton").addEventListener('click',Sortstr);
  </script>
